    <script>
        let selaskiData = null;
        let comexWorkbook = null;

        const inputSelaski = document.getElementById('input-selaski');
        const inputComex = document.getElementById('input-comex');
        const dropSelaski = document.getElementById('drop-selaski');
        const dropComex = document.getElementById('drop-comex');
        const btnProcess = document.getElementById('btn-process');
        const resultPanel = document.getElementById('result-panel');
        const btnDownload = document.getElementById('btn-download');

        dropSelaski.onclick = () => inputSelaski.click();
        dropComex.onclick = () => inputComex.click();

        inputSelaski.onchange = (e) => handleFile(e.target.files[0], 'selaski');
        inputComex.onchange = (e) => handleFile(e.target.files[0], 'comex');

        function showToast(msg) {
            const t = document.getElementById('toast');
            document.getElementById('toast-msg').innerText = msg;
            t.classList.remove('translate-y-32');
            setTimeout(() => t.classList.add('translate-y-32'), 4000);
        }

        function formatDateString(val) {
            if (!val) return "";
            let date;
            if (val instanceof Date) {
                date = val;
            } else if (typeof val === 'number' && val > 30000) {
                date = XLSX.SSF.parse_date_code(val);
                date = new Date(date.y, date.m - 1, date.d);
            } else {
                date = new Date(val);
            }
            if (isNaN(date.getTime())) return val;
            const day = String(date.getDate()).padStart(2, '0');
            const month = String(date.getMonth() + 1).padStart(2, '0');
            const year = date.getFullYear();
            return `${day}-${month}-${year}`;
        }

        async function handleFile(file, type) {
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (e) => {
                const data = new Uint8Array(e.target.result);
                const workbook = XLSX.read(data, { 
                    type: 'array', 
                    cellStyles: true, 
                    cellFormulas: true, 
                    cellDates: true,
                    cellNF: true 
                });

                if (type === 'selaski') {
                    const sheet = workbook.Sheets[workbook.SheetNames[0]];
                    selaskiData = XLSX.utils.sheet_to_json(sheet, { defval: "" });
                    document.getElementById('label-selaski').innerHTML = `<i class="fa-solid fa-check mr-1"></i> ${file.name}`;
                    document.getElementById('label-selaski').classList.replace('bg-slate-100', 'bg-blue-600');
                    document.getElementById('label-selaski').classList.replace('text-slate-600', 'text-white');
                } else {
                    comexWorkbook = workbook;
                    document.getElementById('label-comex').innerHTML = `<i class="fa-solid fa-check mr-1"></i> ${file.name}`;
                    document.getElementById('label-comex').classList.replace('bg-slate-100', 'bg-green-600');
                    document.getElementById('label-comex').classList.replace('text-slate-600', 'text-white');
                }
                if (selaskiData && comexWorkbook) {
                    btnProcess.disabled = false;
                    btnProcess.classList.replace('bg-slate-300', 'bg-blue-600');
                    btnProcess.classList.add('hover:bg-blue-700', 'shadow-blue-200', 'shadow-2xl');
                }
            };
            reader.readAsArrayBuffer(file);
        }

        btnProcess.onclick = () => {
            btnProcess.innerHTML = '<i class="fa-solid fa-cog fa-spin"></i> <span>SINCRONIZANDO...</span>';
            setTimeout(processLogic, 500);
        };

        function processLogic() {
            try {
                const comexSheet = comexWorkbook.Sheets[comexWorkbook.SheetNames[0]];
                
                const mapping = {
                    "Usuario": ["Usuario"],
                    "Nº Embarque": ["N° Exportacion"], 
                    "Contenedores": ["N° Contenedor", "N° Contenedor_1"], 
                    "Puerto Origen": ["Puerto Origen"],
                    "Pais Destino": ["Pais Destino"],
                    "Tipo Contenedor": ["Tipo Contenedor"],
                    "Carrier / Aerolínea": ["Naviera"],
                    "Nave Origen": ["Nave"],
                    "Booking": ["Booking/Reserva"],
                    "Bill of Lading": ["B/L"],
                    "DUS": ["DUS"],
                    "ETD": ["Fecha ETD"],
                    "ETA": ["Fecha ETA"],
                    "N° Factura Exportación": ["N° Factura"],
                    "Fecha Factura Exportación": ["Fecha Factura"],
                    "Valor Factura Exportación": ["Monto Factura"],
                    "Cantidad Despachada": ["Cantidad Despachada"],
                    "Peso Bruto Total": ["KG Brutos Comex"]
                };

                const range = XLSX.utils.decode_range(comexSheet['!ref']);
                let matches = 0;

                const headerMap = {}; 
                for (let C = range.s.c; C <= range.e.c; ++C) {
                    const addr = XLSX.utils.encode_cell({ r: range.s.r, c: C });
                    const cell = comexSheet[addr];
                    if (cell && cell.v) {
                        const headName = cell.v.toString().trim();
                        if (!headerMap[headName]) headerMap[headName] = [];
                        headerMap[headName].push(C);
                    }
                }

                const keyColIndices = headerMap["N° Exportacion"];
                if (!keyColIndices) throw new Error("No se encontró columna 'N° Exportacion'");
                const keyColIdx = keyColIndices[0];

                for (let R = range.s.r + 1; R <= range.e.r; ++R) {
                    const keyAddr = XLSX.utils.encode_cell({ r: R, c: keyColIdx });
                    const keyCell = comexSheet[keyAddr];
                    
                    if (!keyCell || !keyCell.v) continue;

                    const idComex = String(keyCell.v).trim();
                    const sourceMatch = selaskiData.find(s => String(s["Nº Embarque"] || "").trim() === idComex);

                    if (sourceMatch) {
                        matches++;
                        Object.entries(mapping).forEach(([srcColName, destColNames]) => {
                            const val = sourceMatch[srcColName];
                            if (val === undefined || val === "") return;

                            const isDateCol = srcColName.toLowerCase().includes("fecha") || 
                                              srcColName === "ETD" || 
                                              srcColName === "ETA";
                            
                            const finalValue = isDateCol ? formatDateString(val) : val;

                            destColNames.forEach(destName => {
                                const targetIndices = headerMap[destName];
                                if (targetIndices) {
                                    targetIndices.forEach(colIdx => {
                                        const targetAddr = XLSX.utils.encode_cell({ r: R, c: colIdx });
                                        if (comexSheet[targetAddr] && comexSheet[targetAddr].f) return;
                                        if (!comexSheet[targetAddr]) comexSheet[targetAddr] = { t: 's', v: '' };
                                        
                                        if (typeof finalValue === 'number' && !isDateCol) {
                                            comexSheet[targetAddr].t = 'n';
                                            comexSheet[targetAddr].v = finalValue;
                                        } else {
                                            comexSheet[targetAddr].t = 's';
                                            comexSheet[targetAddr].v = String(finalValue);
                                        }
                                    });
                                }
                            });
                        });
                    }
                }

                document.getElementById('count-matches').innerText = matches;
                resultPanel.classList.remove('hidden');
                btnProcess.innerHTML = '<span>Sincronizar Nuevamente</span> <i class="fa-solid fa-rotate"></i>';
                showToast("Sincronización finalizada");

            } catch (err) {
                console.error(err);
                showToast("Error: " + err.message);
                btnProcess.innerHTML = '<span>Error en proceso</span>';
            }
        }

        btnDownload.onclick = () => {
            XLSX.writeFile(comexWorkbook, "Plantilla_Comex_Actualizada.xlsx", { 
                bookType: 'xlsx', 
                cellStyles: true,
                cellFormulas: true 
            });
        };

    </script>
